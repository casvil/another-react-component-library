# Another React Component Library

A customizable React 19 component library built with Vite, Tailwind CSS v4 and TypeScript.

## üöÄ Getting Started

Install the component library in your React project:

```bash
npm install another-react-component-library
```

### Basic Usage

```tsx
import {
  Form,
  FormInput,
  Select,
  Switch,
  Clock,
} from 'another-react-component-library';

const fruitOptions = [
  { value: 'apple', label: 'üçé Apple' },
  { value: 'banana', label: 'üçå Banana' },
  { value: 'orange', label: 'üçä Orange' },
  { value: 'grape', label: 'üçá Grape' },
];

function App() {
  return (
    <div>
      <div style={{ display: 'flex', justifyContent: 'center' }}>
        <Clock format="HH:mm:ss" bordered />
      </div>
      <div>
        <Form onSubmit={(data) => alert(JSON.stringify(data, null, 2))}>
          <FormInput
            name="username"
            label="Username"
            placeholder="Enter your username"
            required
          />
          <Select
            label="Favorite Fruit"
            options={fruitOptions}
            defaultValue="apple"
          />
          <Switch
            label="Enable notifications"
            defaultChecked={true}
            wrapperClassName="space-x-6"
          />
        </Form>
      </div>
    </div>
  );
}
```
